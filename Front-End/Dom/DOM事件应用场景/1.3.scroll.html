Get div height with plain JavaScript
var clientHeight = document.getElementById('myDiv').clientHeight; or var offsetHeight = document.getElementById('myDiv').offsetHeight; clientHeight includes padding. offsetHeight includes padding, scrollBar and borders.
Scrolling to the bottom inside a div triggers an alert (Pure JavaScript)
document.querySelector('#a').addEventListener('scroll',scrollTrigger); function scrollTrigger(e) { const a = e.currentTarget; // use greater than or equal to be sure. if (Math.ceil(a.scrollTop) + a.clientHeight >= a.scrollHeight) { alert('End of scroll'); } }
